      383 +def best_damage_through(                                      
      384 +    attackers_details: list[dict],                            
      385 +    blockers_details: list[dict],                             
      386 +) -> int:                                                     
      387 +    """Greedy estimate of how much damage gets through after o
          +ptimal blocks.                                                
      388 +                                                              
      389 +    Both sides have full P/T. We pair attackers with blockers 
          +greedily:                                                     
      390 +    blockers volunteer to chump or trade; the attacker side pi
          +cks the order                                                 
      391 +    that maximizes damage through. This is a coarse upper-boun
          +d estimate but                                                
      392 +    avoids the false "1 attacker = lethal in 20 turns" trap wh
          +en there's a                                                  
      393 +    real board stall.                                         
      394 +    """                                                       
      395 +    attackers = sorted(                                       
      396 +        [a for a in attackers_details if not a.get("tapped")],
      397 +        key=lambda a: -a.get("power", 0),                     
      398 +    )                                                         
      399 +    blockers = sorted(                                        
      400 +        [b for b in blockers_details if not b.get("tapped")], 
      401 +        key=lambda b: -b.get("toughness", 0),                 
      402 +    )                                                         
      403 +    if not attackers:                                         
      404 +        return 0                                              
      405 +    available_blockers = list(blockers)                       
      406 +    damage = 0                                                
      407 +    for atk in attackers:                                     
      408 +        if not available_blockers:                            
      409 +            damage += atk["power"]                            
      410 +            continue                                          
      411 +        # Best blocker = smallest one whose toughness >= attac
          +ker's power,                                                  
      412 +        # falling back to the largest blocker for a chump.    
      413 +        survivors = [b for b in available_blockers if b["tough
          +ness"] > atk["power"]]                                        
      414 +        if survivors:                                         
      415 +            # Trade or eat the attacker; no damage through.   
      416 +            blocker = min(survivors, key=lambda b: b["toughnes
          +s"])                                                          
      417 +            available_blockers.remove(blocker)                
      418 +            continue                                          
      419 +        # Chump with the smallest blocker — attacker still sur
          +vives.                                                        
      420 +        chump = min(available_blockers, key=lambda b: b["power
          +"])                                                           
      421 +        available_blockers.remove(chump)                      
      422 +        # If the chump is bigger than attacker's toughness, it
          + trades; no damage.                                           
      423 +        if chump["power"] >= atk["toughness"]:                
      424 +            continue                                          
      425 +        damage += atk["power"]                                
      426 +    return damage
